Draws this circle into the provided CanvasRenderingContext2D

void draw( CanvasRenderingContext2D context ) {
  context.save();
  context.fillStyle = color;
  context.strokeStyle = '';
  context.beginPath();
  context.arc(x, y, radius, 0, TAU, false);
  context.fill();
  context.closePath();
}
